// Цикл по входным кодам
while (больше данных для чтения) {
    // Чтение очередного кода
    current_code = чтение очередного кода из входных данных

    // Если текущий код есть в словаре, то получаем
       соответствующую ему последовательность символов
    if (current_code уже есть в словаре) {
        current = символы, соответствующие current_code в словаре
    }
    // Иначе текущая последовательность символов - это предыдущая
       последовательность символов с добавлением ее первого символа
    // И добавляем эту последовательность в словарь
    else {
        current = символ, соответствующий previous_code в словаре,
                  исключая последний символ
        current = current + первый символ, соответствующий current_code в словаре
        добавление текущей последовательности и ее кода в словарь
    }

    // Вывод текущей последовательности в выходной поток
    запись current в выходной поток

    // Обновление переменных для следующей итерации цикла
    previous = текущая последовательность
    previous_code = текущий код
}
